    c.name,

    COUNT(o.id) as order_count,
    SUM(o.amount) as total_revenue,
    AVG(o.amount) as avg_order_value
FROM customers c
LEFT JOIN orders o ON c.id = o.customer_id
WHERE o.date >= CURRENT_DATE - INTERVAL '90 days'
